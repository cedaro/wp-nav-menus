function twentysixteen_scripts() {
	wp_enqueue_script(
		'twentysixteen-nav-menus',
		get_theme_file_uri( 'js/wp-nav-menus.js' ),
		array(),
		'1.0.0',
		true
	);

	wp_localize_script( 'twentysixteen-nav-menus', '_cedaroNavMenuL10n', array(
		'collapseSubmenu' => __( 'collapse child menu', 'twentysixteen' ),
		'expandSubmenu'   => __( 'expand child menu', 'twentysixteen' ),
	) );
}
add_action( 'wp_enqueue_scripts', 'twentysixteen_scripts' );


$( '.menu-main-menu-container' ).cedaroNavMenu({
	activeMenuItemClass: 'focus',
	breakpoint: 910,
	expandedSubmenuClass: 'toggled-on',
	expandedSubmenuToggleClass: 'toggled-on',
	submenuToggleClass: 'dropdown-toggle'
});
