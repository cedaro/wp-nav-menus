function twentyseventeen_scripts() {
	wp_enqueue_script(
		'twentyseventeen-nav-menus',
		get_theme_file_uri( 'assets/js/wp-nav-menus.js' ),
		array(),
		'1.0.0',
		true
	);

	wp_localize_script( 'twentyseventeen-nav-menus', '_cedaroNavMenuL10n', array(
		'collapseSubmenu' => __( 'Collapse child menu', 'twentyseventeen' ),
		'expandSubmenu'   => __( 'Expand child menu', 'twentyseventeen' ),
	) );
}
add_action( 'wp_enqueue_scripts', 'twentyseventeen_scripts' );


$( '.menu-main-menu-container' )
	.cedaroNavMenu({
		activeMenuItemClass: 'focus',
		breakpoint: 768,
		expandedSubmenuClass: 'toggled-on',
		expandedSubmenuToggleClass: 'toggled-on',
		submenuToggleClass: 'dropdown-toggle'
	})
	.find( '.dropdown-toggle' )
	.prepend( twentyseventeenScreenReaderText.icon );
