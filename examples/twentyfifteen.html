function twentyfifteen_scripts() {
	wp_enqueue_script(
		'twentyfifteen-nav-menus',
		get_theme_file_uri( 'js/wp-nav-menus.js' ),
		array(),
		'1.0.0',
		true
	);

	wp_localize_script( 'twentyfifteen-nav-menus', '_cedaroNavMenuL10n', array(
		'collapseSubmenu' => __( 'collapse child menu', 'twentyfifteen' ),
		'expandSubmenu'   => __( 'expand child menu', 'twentyfifteen' ),
	) );
}
add_action( 'wp_enqueue_scripts', 'twentyfifteen_scripts' );


$( '.menu-main-menu-container' ).cedaroNavMenu({
	breakpoint: 955,
	expandCurrentItem: true,
	expandedSubmenuClass: 'toggled-on',
	expandedSubmenuToggleClass: 'toggle-on',
	submenuToggleClass: 'dropdown-toggle',
	submenuToggleMode: 'click'
});
